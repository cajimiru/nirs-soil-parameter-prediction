	\textbf{function} $\m{nbr}(M)$\{\\
		\>$r = \m{rnd}\set{1,\ldots,k}$\\
		\\
		\>\textbf{if} ($\lambda_r\in M$)\{\\
			\>\>$\widetilde{M}=M\setminus\set{\lambda_r}$\\
		\>\}\textbf{else}\{\\
			\>\>$\widetilde{M}=M\cup\set{\lambda_r}$\\
		\>\}\\
		\\
		\>\textbf{return} $\widetilde{M}$\\
	\}
